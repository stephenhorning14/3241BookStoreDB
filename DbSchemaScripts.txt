CREATE TABLE "Author" (
	"AuthorNum"	INTEGER NOT NULL,
	"Name"	TEXT NOT NULL,
	PRIMARY KEY("AuthorNum")
);

CREATE TABLE "Employee" (
	"EmployeeNum"	TEXT NOT NULL,
	"Address"	TEXT NOT NULL,
	"Name"	TEXT NOT NULL,
	PRIMARY KEY("EmployeeNum")
);

CREATE TABLE "Publisher" (
	"Name"	TEXT NOT NULL,
	"Address"	TEXT NOT NULL,
	PRIMARY KEY("Name")
);

CREATE TABLE "Supplier" (
	"SupplierNum"	INTEGER NOT NULL,
	"Name"	TEXT NOT NULL,
	"Address"	TEXT NOT NULL,
	PRIMARY KEY("SupplierNum")
);

CREATE TABLE "Customer" (
	"AccountNum"	INTEGER NOT NULL,
	"Name"	TEXT NOT NULL,
	"Birthday"	TEXT NOT NULL,
	"Address"	TEXT NOT NULL,
  "EmailAddress" TEXT,
	PRIMARY KEY("AccountNum")
);

CREATE TABLE "Item" (
	"ItemNum"	INTEGER NOT NULL,
	"SellPrice"	REAL NOT NULL,
	"PurchasedPrice"	REAL NOT NULL,
	"Quantity"	INTEGER NOT NULL,
	PRIMARY KEY("ItemNum")
);

CREATE TABLE "Book" (
	"ISBN"	TEXT NOT NULL,
	"Title"	TEXT NOT NULL,
	"Genre"	TEXT NOT NULL,
	"ReleaseYear"	TEXT NOT NULL,
	"ItemNum"	INTEGER,
	"Publisher"	TEXT NOT NULL,
	FOREIGN KEY("ItemNum") REFERENCES "Item"("ItemNum"),
	PRIMARY KEY("ISBN"),
	FOREIGN KEY("Publisher") REFERENCES "Publisher"("Name")
);

CREATE TABLE "CD" (
	"ItemNum"	INTEGER NOT NULL,
	"Artist"	TEXT NOT NULL,
	"Genre"	TEXT NOT NULL,
	"ReleaseDate"	TEXT NOT NULL,
	PRIMARY KEY("ItemNum"),
	FOREIGN KEY("ItemNum") REFERENCES "Item"("ItemNum")
);

CREATE TABLE "DVD" (
	"ItemNum"	INTEGER NOT NULL,
	"Director"	TEXT NOT NULL,
	"ReleaseDate"	TEXT NOT NULL,
	"Title"	TEXT NOT NULL,
	"Format"	TEXT NOT NULL,
	"Genre"	TEXT NOT NULL,
	PRIMARY KEY("ItemNum"),
	FOREIGN KEY("ItemNum") REFERENCES "Item"("ItemNum")
);

CREATE TABLE "Purchase" (
	"PurchaseNum"	INTEGER NOT NULL,
	"Date"	TEXT NOT NULL,
	"TotalPrice"	REAL NOT NULL,
	"PaymentMethod"	TEXT NOT NULL,
	"SoldBy"	INTEGER NOT NULL,
	"SoldTo"	INTEGER NOT NULL,
	FOREIGN KEY("SoldTo") REFERENCES "Customer"("AccountNum"),
	FOREIGN KEY("SoldBy") REFERENCES "Employee"("EmployeeNum"),
	PRIMARY KEY("PurchaseNum")
);

CREATE TABLE "PurchasedItems" (
	"ItemNum"	INTEGER NOT NULL,
	"PurchaseNum"	INTEGER NOT NULL,
	PRIMARY KEY("ItemNum","PurchaseNum"),
	FOREIGN KEY("ItemNum") REFERENCES "Item"("ItemNum"),
	FOREIGN KEY("PurchaseNum") REFERENCES "Purchase"("PurchaseNum")
);

CREATE TABLE "AuthoredBooks" (
	"AuthorNum"	INTEGER NOT NULL,
	"ISBN"	TEXT NOT NULL,
	FOREIGN KEY("AuthorNum") REFERENCES "Author"("AuthorNum"),
	FOREIGN KEY("ISBN") REFERENCES "Book"("ISBN"),
	PRIMARY KEY("AuthorNum","ISBN")
);

CREATE TABLE "SuppliedItems" (
	"ItemNum"	INTEGER NOT NULL,
	"SupplierNum"	INTEGER NOT NULL,
	FOREIGN KEY("ItemNum") REFERENCES "Item"("ItemNum"),
	FOREIGN KEY("SupplierNum") REFERENCES "Supplier"("SupplierNum"),
	PRIMARY KEY("ItemNum","SupplierNum")
);
